serotype_labeller <- function(variable) {
    total_count <- table(samples_df$serotype)[variable]
    paste0(variable, " (n = ", total_count, ")")
}

nice_expansion <- expansion(add = 0, mult = c(0, 0.1))

plate_labeller <- function(plate_string) {
    plate_date <- format(as.Date(str_replace(plate_string, "_72h", ""), format = "%d%b%y"), "%b. %d, %Y")
    this_label <- str_sub(plate_string, -1)
    plate_date <- case_when(
        this_label %in% c("A", "B", "C") ~ paste(plate_date, this_label),
        TRUE ~ plate_date
    )

    return(plate_date)
}

categorize_biofilm_formation <- function(cv_reading, sd_vbe, base = 0) {
    case_when(
        cv_reading < base + sd_vbe ~ "non-former",
        between(cv_reading, base + sd_vbe, base + 2 * sd_vbe) ~ "weak",
        between(cv_reading, base + 2 * sd_vbe, base + 4 * sd_vbe) ~ "moderate",
        cv_reading > base + 4 * sd_vbe ~ "strong"
    )
}

Calculating per-plate VBE averages

From Shawn’s protocol:

Determine biofilm production using three standard deviations above the mean of the negative OD (OD cut-off, or ODc). A non-biofilm producer is defined as OD < ODc; weak as ODc < OD < 2 x ODc; moderate as 2 x ODc < OD < 4 x ODc; and strong as OD > 4 x ODc.

Code 
vbe_summaries <- sbfp_data %>%
    filter(source_id == "VBE") %>%
    filter(!is.na(cv_1_od538)) %>%
    mutate(mean_cv = (cv_1_od538 + cv_2_od538) / 2) %>%
    group_by(x72_hr_plate) %>%
    summarise(
        mean_vbe = mean(mean_cv),
        sd_vbe = sd(mean_cv),
        n_vbe = n()
    ) %>%
    mutate(
        plate_label = plate_labeller(x72_hr_plate),
        plate_label = paste0(plate_label, " (n = ", n_vbe, ")"),
    )

We’ll test for significant differences in mean OD by serotype using ANOVA.

ANOVA 
anova_results <- aov(
    mean ~ serotype,
    data = samples_df
)

ANOVA shows that mean OD significantly differs based on Serotype (\(p \approx 4.41e-10\)). We’ll use a Tukey post-hoc test to determine where these significant differences are. Since TukeyHSD already does multiple testing correction, we’ll just use \(\alpha = 0.05\) here.

TukeyHSD 
tukey_results <- TukeyHSD(anova_results)

p_thresh <- 0.05

tukey_results$serotype %>%
    as.data.frame() %>%
    janitor::clean_names() %>%
    rownames_to_column("contrast") %>%
    filter(p_adj < p_thresh) %>%
    select(contrast, p_adj)
